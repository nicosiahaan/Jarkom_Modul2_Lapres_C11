### 1.	Membuat semeruc11.pw pada UML Malang
-	nano /etc/bind/named.conf.local
-	mkdir /etc/bind/jarkom
-	cp /etc/bind/db.local /etc/bind/jarkom/semeruc11.pw
-	nano /etc/bind/jarkom/semeruc11.pw
-	service bind9 restart
